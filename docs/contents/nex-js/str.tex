\section{String}
\label{String}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]]
		The following functions enhance string processing capabilities, introducing features for sanitization, character extraction, and conditional formatting.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{Str.join(val, str, sep = '')}: Concatenates the given string \NexFunction{str} to the value \NexFunction{val}, using the optional separator \NexFunction{sep}. Ensures \NexFunction{val} is defined before appending the separator and the string.}
			\NexItemDark{\NexFunction{Str.merge(val1, val2, def1 = ' ', def2 = ' ', sep = ' ')}: Combines two values \NexFunction{val1} and \NexFunction{val2}, defaulting to \NexFunction{def1} and \NexFunction{def2} if undefined. Uses \NexFunction{sep} as the separator and relies on \NexFunction{Str.join} for the final concatenation.}
			\NexItemDark{\NexFunction{Str.toLower(val)}: Converts the given value \NexFunction{val} to lowercase. Ensures \NexFunction{val} is defined before transformation. Returns the transformed string.}
			\NexItemDark{\NexFunction{Str.toUpper(val)}: Converts the given value \NexFunction{val} to uppercase. Ensures \NexFunction{val} is defined before performing the transformation. Returns the transformed string.}
			\NexItemDark{\NexFunction{Str.camelCase(val)}: Transforms the given string \NexFunction{val} into camel case format. Adjusts capitalization based on position and removes unnecessary spaces or special characters, ensuring the first letter is lowercase.}
			\NexItemDark{\NexFunction{Str.titleCase(val)}: Converts the given string \NexFunction{val} into title case format. Capitalizes the first letter of each word and lowers the rest, separating words with spaces for consistent formatting.}
			\NexItemDark{\NexFunction{Str.camelTitleCase(val)}: Converts a camel-case formatted string \NexFunction{val} into a readable title case format. Splits the string into words based on uppercase letters and ensures proper capitalization of the first word.}
			\NexItemDark{\NexFunction{Str.joinCase(val, rpl = '', fnd = '\\s+')}: Joins segments of the given value \NexFunction{val} using a replacement string \NexFunction{rpl}. Splits the input based on the provided delimiter \NexFunction{fnd}, transforming each segment to lowercase before joining.}
			\NexItemDark{\NexFunction{Str.kebabCase(val)}: Converts the given string \NexFunction{val} into kebab-case format. Uses \NexFunction{Str.joinCase} to replace spaces with hyphens and ensures proper formatting.}
			\NexItemDark{\NexFunction{Str.snakeCase(val)}: Converts the given string \NexFunction{val} into snake-case format by replacing spaces or delimiters with underscores (\texttt{\_}). Uses \NexFunction{Str.joinCase} for consistent processing.}
			\NexItemDark{\NexFunction{Str.remove(val, mth = 'null|undefined')}: Cleans the given string \NexFunction{val} by removing matches for the pattern \NexFunction{mth}, replacing them with spaces. Normalizes and trims the result to ensure no excess whitespace remains.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}

\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secB]
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{Str.upper()}: Returns a string containing all uppercase English alphabet letters (\texttt{A-Z}). Does not require any input and ensures consistent output.}
			\NexItemDark{\NexFunction{Str.lower()}: Returns a string containing all lowercase English alphabet letters (\texttt{a-z}). Does not require any input and ensures consistent output.}
			\NexItemDark{\NexFunction{Str.digit()}: Returns a string containing all numeric digits (\texttt{0-9}). Designed for easy inclusion in character sets.}
			\NexItemDark{\NexFunction{Str.xdigit(val)}: Generates a string of hexadecimal characters based on the given value \NexFunction{val}. Includes lowercase (\texttt{a-f}), uppercase (\texttt{A-F}), and digits (\texttt{0-9}). Adjusts the inclusion of uppercase or lowercase letters based on truthiness or falsiness of \NexFunction{val}.}
			\NexItemDark{\NexFunction{Str.punct()}: Returns a string containing common punctuation characters, including symbols such as \texttt{!}, \texttt{\#}, \texttt{@}, and more.}
			\NexItemDark{\NexFunction{Str.alpha()}: Generates a string containing all alphabetic characters, combining uppercase (\texttt{A-Z}) and lowercase (\texttt{a-z}) alphabets.}
			\NexItemDark{\NexFunction{Str.alnum()}: Combines alphabetic (\NexFunction{Str.alpha()}) and numeric (\NexFunction{Str.digit()}) characters into a single string.}
			\NexItemDark{\NexFunction{Str.graph()}: Produces a string containing all printable graphical characters, combining alphanumeric (\NexFunction{Str.alnum()}) and punctuation (\NexFunction{Str.punct()}) characters.}
			\NexItemDark{\NexFunction{Str.random(val, chars = 'alnum', sep = ',')}: Generates a random string of length \NexFunction{val} using the specified character sets \NexFunction{chars}. Supports various character options such as \texttt{alnum}, \texttt{digit}, \texttt{graph}, \texttt{lower}, and more. Allows customization by splitting character definitions with the provided separator \NexFunction{sep}.}
			\NexItemDark{\NexFunction{Str.implode(val, sep = '')}: Joins the elements of the array \NexFunction{val} into a single string, using the optional separator \NexFunction{sep}. Returns an empty string if \NexFunction{val} is not an array.}
			\NexItemDark{\NexFunction{Str.sanitize(val)}: Cleans the given string \NexFunction{val} by removing all non-alphanumeric characters (\texttt{a-z}, \texttt{A-Z}, \texttt{0-9}). Returns the sanitized string.}
			\NexItemDark{\NexFunction{Str.lastChar(str, val)}: Extracts the portion of the string \NexFunction{str} following the last occurrence of the specified value \NexFunction{val}. Ensures both inputs are defined before performing the operation. Returns the resulting substring or \texttt{undefined} if \NexFunction{val} is not found.}
			\NexItemDark{\NexFunction{Str.conditional(val, pre = '', post = '')}: Constructs a new string by appending the prefix \NexFunction{pre} and suffix \NexFunction{post} around the value \NexFunction{val}. Returns an empty string if \NexFunction{val} is not defined.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
