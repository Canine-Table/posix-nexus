\section{Event}
\label{Event}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]]
		The following functions provide advanced event handling utilities, enabling robust management of DOM events, mutations, overflow detection, and controlled execution timing.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{nexEvent.add(obj)}: Adds event listeners to a specified DOM node. Maps shorthand event identifiers to their full names (e.g., \texttt{'c'} to \texttt{'click'}) and applies the associated action for each event. Validates event and node details before execution.}
			\NexItemDark{\NexFunction{nexEvent.mutation(ml, obs)}: Handles mutation events based on a list \NexFunction{ml}. Logs changes to child nodes or attributes, such as additions, removals, or modifications, with detailed outputs.}
			\NexItemDark{\NexFunction{nexEvent.overflow(elm)}: Determines whether the content of the given element \NexFunction{elm} exceeds its visible dimensions, checking both height and width overflow. Returns \texttt{true} if overflow exists, otherwise \texttt{false}.}
			\NexItemDark{\NexFunction{nexEvent.throttle(func, limit)}: Restricts the execution frequency of the given function \NexFunction{func} to once every \NexFunction{limit} milliseconds. Ensures proper spacing of executions, even during rapid triggering.}
			\NexItemDark{\NexFunction{nexEvent.debounce(func, delay)}: Delays the execution of the given function \NexFunction{func} until \NexFunction{delay} milliseconds have passed since the last invocation. Helps reduce redundant calls during high-frequency triggers.}
			\NexItemDark{\NexFunction{nexEvent.delay(ms)}: Creates a delay for the specified number of milliseconds \NexFunction{ms}. Returns a promise that resolves after the delay, allowing asynchronous control.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
