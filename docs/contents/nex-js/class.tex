\section{Class}
\label{Class}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]]
		The following functions offer versatile utilities for managing and manipulating object classes, enabling default property handling, aliasing, and hierarchical navigation.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{nexClass.defaults(def, obj, mth)}: Assigns default properties from \NexFunction{def} to the given object \NexFunction{obj}. Utilizes \NexFunction{mth} for method-based value matching. Removes properties with undefined or \texttt{false} values while populating the \texttt{obj.prop} structure.}
			\NexItemDark{\NexFunction{nexClass.prop(ref, obj, cls)}: Maps properties from the reference array \NexFunction{ref} to the given object \NexFunction{obj}. Supports nested property handling and invokes class-specific functions on matching elements when the class \NexFunction{cls} is provided.}
			\NexItemDark{\NexFunction{nexClass.expand(val, obj)}: Expands the object \NexFunction{val} into \NexFunction{obj} by converting array values into key-value mappings, preserving object relationships.}
			\NexItemDark{\NexFunction{nexClass.cherryPick(obj)}: Randomly selects and returns an element from the array \NexFunction{obj}. Uses \NexFunction{Int.loop()} for cyclic wrapping of the index.}
			\NexItemDark{\NexFunction{nexClass.alias(als, opt, obj)}: Applies alias mappings defined in \NexFunction{als} to the object \NexFunction{obj}, replacing keys with their corresponding aliases. Supports optional value matching using \NexFunction{opt}. Updates the object structure dynamically.}
			\NexItemDark{\NexFunction{nexClass.back(obj, num = 1)}: Recursively traverses the parent hierarchy of the object \NexFunction{obj}, moving up by \NexFunction{num} levels. Returns the resulting ancestor object or the original \NexFunction{obj} if no further traversal is possible.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
