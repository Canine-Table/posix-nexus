\begin{NxSSBox}[breakable][Automata Theory]
	\begin{NxIDBox}
		Automata Theory explores mathematical models of computation that process input based on predefined rules. These models define system behavior through states, transitions, and acceptance conditions.
	\end{NxIDBox}
	\begin{NxIDBoxL}
		\nxTopicD{Finite-State Automata (FSA)} Models that use a finite set of states to process input sequences deterministically or non-deterministically.
		%\nxTopicD{Pushdown Automata (PDA)} Automata that introduce a stack memory, allowing recognition of context-free languages.
		%\nxTopicD{Turing Machines (TM)} Abstract machines capable of simulating any algorithm, defining the theoretical limits of computation.
		%\nxTopicD{Formal Language Hierarchy} Classification of languages based on computational complexity and recognizability.
		%\nxTopicD{Decidability and Complexity} Investigation into which problems can be solved within different automata models and their computational efficiency.
	\end{NxIDBoxL}
\end{NxSSBox}
