\section{Str}
\label{Str}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]
		The following functions enable dynamic string manipulation, including random string generation, targeted searches and replacements, and case adjustments for versatile text handling.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{get\_str\_rand()}: Generates a random string of specified length \NexFunction{num}, using character sets \NexFunction{chars} such as alphanumeric or others.}
			\NexItemDark{\NexFunction{get\_str\_locate()}: Searches for occurrences of a string \NexFunction{fnd} within the input, optionally replacing it with \NexFunction{rpl}, separating content with \NexFunction{sep}, and supporting global or targeted searches.}
			\NexItemDark{\NexFunction{set\_str\_case()}: Modifies the case of a string, converting it to uppercase, lowercase, or title case, based on the provided option (\texttt{u}, \texttt{l}, or \texttt{t}).}
			\NexItemDark{\NexFunction{set\_str\_format()}: Formats a string based on a specified format \NexFunction{fmt}, with optional separators \NexFunction{sep}, and alignment options like \NexFunction{lft}, \NexFunction{rgt}, or \NexFunction{kp}.}
			\NexItemDark{\NexFunction{add\_str\_append()}: Appends a specified character \NexFunction{char} or string multiple times to reach a desired length \NexFunction{num}, optionally extending or modifying the input based on \NexFunction{ed}.}
			\NexItemDark{\NexFunction{add\_str\_div()}: Creates a horizontal divider string of the length derived from terminal column size, using repeated characters like \texttt{"-"}.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
