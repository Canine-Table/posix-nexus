\section{Networking}
\label{Networking}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]]
		The following functions provide utilities for working with Layer 2 and Layer 3 network addresses and types, including address generation, validation, and classification.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{\_\_get\_net\_virt\_types()}: Provides a comma-separated list of supported virtual network device types for validation and reference.}
			\NexItemDark{\NexFunction{\_\_chk\_net\_virt\_type()}: Validates the given virtual network type against the supported types returned by \NexFunction{\_\_get\_net\_virt\_types()}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_name()}: Retrieves a comma-separated list of system network interface names using \texttt{ip link show}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_alt()}: Extracts alternative names (altnames) for network interfaces as a comma-separated list from \texttt{ip link show}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_names()}: Combines the outputs of \NexFunction{\_\_get\_net\_dev\_alt()} and \NexFunction{\_\_get\_net\_dev\_name()} into a unified comma-separated list of all interface names.}
			\NexItemDark{\NexFunction{\_\_chk\_net\_dev\_names()}: Validates the given network interface name against the combined list of names produced by \NexFunction{\_\_get\_net\_dev\_names()}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_realname()}: Retrieves the real name of a network device based on the given alias, using \texttt{ip link show} and validation via \NexFunction{\_\_chk\_net\_dev\_names()}.}
			\NexItemDark{\NexFunction{\_\_is\_net\_dev()}: Checks whether the given device name corresponds to a valid network interface directory under \texttt{/sys/class/net}, using \NexFunction{\_\_get\_net\_dev\_realname()}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_obj()}: Provides a comma-separated list of supported network object types, such as addresses, routes, and tunnels.}
			\NexItemDark{\NexFunction{\_\_chk\_net\_obj()}: Validates the given network object type against the list returned by \NexFunction{\_\_get\_net\_obj()}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_fam()}: Provides a comma-separated list of supported network families, including \texttt{inet}, \texttt{inet6}, and others.}
			\NexItemDark{\NexFunction{\_\_chk\_net\_fam()}: Validates the given network family against the list produced by \NexFunction{\_\_get\_net\_fam()}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_ipv6gen()}: Provides a comma-separated list of supported IPv6 address generation methods, such as \texttt{eui64} and \texttt{random}.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}

\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secB]
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{\_\_chk\_net\_dev\_ipv6gen()}: Validates the given IPv6 address generation method against the list produced by \NexFunction{\_\_get\_net\_dev\_ipv6gen()}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_files()}: Lists all files associated with a given network device by iterating over its directory in \texttt{/sys/class/net}.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_list()}: Lists all network devices in \texttt{/sys/class/net} that are symbolic links, extracting their names.}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_file()}: Retrieves the content of a specific file for a given network device in \texttt{/sys/class/net}, returning \texttt{2} if the file does not exist.}
			\NexItemDark{\NexFunction{get\_net\_dev\_alias()}: Retrieves the alias of a specified network device by reading its \texttt{ifalias} file in \texttt{/sys/class/net}.}
			\NexItemDark{\NexFunction{get\_net\_dev\_index()}: Retrieves the index of a specified network device by reading its \texttt{ifindex} file.}
			\NexItemDark{\NexFunction{get\_net\_dev\_duplex()}: Retrieves the duplex mode of a specified network device by reading its \texttt{duplex} file.}
			\NexItemDark{\NexFunction{get\_net\_dev\_state()}: Retrieves the operational state of a specified network device by reading its \texttt{operstate} file.}
			\NexItemDark{\NexFunction{get\_net\_dev\_mtu()}: Retrieves the MTU (Maximum Transmission Unit) of a specified network device by reading its \texttt{mtu} file.}
			\NexItemDark{\NexFunction{get\_net\_dev\_speed()}: Retrieves the speed of a specified network device by reading its \texttt{speed} file.}
			\NexItemDark{\NexFunction{get\_net\_dev\_l2()}: Determines the Layer 2 type of a network device (e.g., \texttt{ether}, \texttt{loopback}) using \texttt{ip address show}.}
			\NexItemDark{\NexFunction{get\_net\_dev\_qlen()}: Retrieves the transmit queue length (\texttt{tx\_queue\_len}) of a network device by reading its respective file.}
			\NexItemDark{\NexFunction{get\_net\_dev\_inet6()}: Extracts the IPv6 address details of a network device using \texttt{ip address show}.}
			\NexItemDark{\NexFunction{get\_net\_dev\_inet()}: Retrieves the first IPv4 or IPv6 address of a specified network device using \texttt{ip address show}.}
			\NexItemDark{\NexFunction{get\_net\_dev\_inet4()}: Extracts the first IPv4 address of a specified network device by filtering the output of \texttt{ip address show}.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}

\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secB]
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{\_\_get\_net\_dev\_info()}: Constructs network device information options dynamically based on command-line arguments. Options include network family, object type, verbosity, and device type, enabling flexible data processing.}
			\NexItemDark{\NexFunction{get\_net\_l3\_type()}: Determines the Layer 3 type (\texttt{inet} or \texttt{inet6}) of a given address, using Awk to validate against IPv4 and IPv6 standards.}
			\NexItemDark{\NexFunction{get\_net\_l2\_type()}: Identifies the Layer 2 type of a given address using Awk and returns its classification if valid.}
			\NexItemDark{\NexFunction{new\_net\_l2\_address()}: Generates a new Layer 2 address, with options to customize universal/local, unicast/multicast, and formatting preferences. Uses Awk for validation and processing.}
			\NexItemDark{\NexFunction{get\_net\_eui64()}: Generates an IPv6 EUI-64 address from a provided Layer 2 address. Offers customization for separators and casing (uppercase or lowercase), using Awk for validation and processing.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
