\section{Cmd}
\label{Cmd}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]
		The following functions offer essential utilities for discovering and verifying the availability of commands, enhancing the portability and adaptability of shell scripts across different environments.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{get\_cmd()}: Iterates through a list of commands provided as arguments, checks their availability using \texttt{command -v}, and returns the first found command or exits if none are found.}
			\NexItemDark{\NexFunction{get\_cmd\_pager()}: Searches for commonly used pager commands (\texttt{less}, \texttt{more}, and \texttt{tee}) by leveraging the \NexFunction{get\_cmd()} function.}
			\NexItemDark{\NexFunction{get\_cmd\_awk()}: Searches for AWK implementations (\texttt{mawk}, \texttt{nawk}, \texttt{awk}, \texttt{gawk}) using the \NexFunction{get\_cmd()} function.}
			\NexItemDark{\NexFunction{get\_cmd\_shell()}: Searches for available shell interpreters (\texttt{dash}, \texttt{sh}, \texttt{bash}, \texttt{zsh}, \texttt{fish}, and others) in the current environment.}
			\NexItemDark{\NexFunction{get\_cmd\_editor()}: Locates command-line text editors (\texttt{nvim}, \texttt{vim}, \texttt{gvim}, \texttt{vi}) for editing files.}
			\NexItemDark{\NexFunction{get\_cmd\_tex\_compiler()}: Searches for LaTeX compilation utilities (\texttt{latexmk}, \texttt{pdflatex}, \texttt{lualatex}, \texttt{xelatex}).}
			\NexItemDark{\NexFunction{get\_cmd\_pdf\_viewer()}: Finds installed PDF viewers (\texttt{zathura}, \texttt{mupdf}, \texttt{evince}).}
			\NexItemDark{\NexFunction{get\_cmd\_pkgmgr()}: Searches for package management tools (\texttt{pacman}, \texttt{apt}, \texttt{dnf}, \texttt{brew}, and others) in the system.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
