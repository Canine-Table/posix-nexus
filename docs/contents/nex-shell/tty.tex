\section{Tty}
\label{Tty}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]
		The following functions provide utilities for managing and querying TTY properties, including property retrieval, structured formatting, and signal handling for enhanced user interaction.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{get\_tty\_prop\_list()}: Lists all TTY properties in key-value pairs, processing the output of \texttt{stty -a} for structured formatting.}
			\NexItemDark{\NexFunction{get\_tty\_prop()}: Retrieves specific TTY properties based on provided keys (\texttt{-k}) or values (\texttt{-v}), enabling focused property queries.}
			\NexItemDark{\NexFunction{set\_tty\_hault()}: Temporarily disables the cursor using \texttt{setterm} and traps signals to re-enable it upon script exit or interruption.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
