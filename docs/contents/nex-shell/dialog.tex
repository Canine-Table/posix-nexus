\section{Dialog}
\label{Dialog}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]]
		The following functions enable dynamic, customizable dialog interactions for time management, user confirmations, and various structured inputs.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{\_\_get\_dialog\_factory()}: Constructs dialog window definitions dynamically, based on specified options (\texttt{-v}, \texttt{-m}, \texttt{-b}, \texttt{-p}, \texttt{-e}). Supports multiple dialog types, argument parsing, and output formatting.}
			\NexItemDark{\NexFunction{\_\_get\_dialog\_output(opt, fld, dft)}: Processes dialog options \NexFunction{opt} and fields \NexFunction{fld} to generate a formatted output string. Uses auxiliary Awk scripts for operations like trimming, splitting, and retrieving default values \NexFunction{dft} when specified. Escapes non-alphanumeric characters in field names.}
			\NexItemDark{\NexFunction{\_\_get\_dialog\_selected(opt, fld)}: Identifies selected fields from dialog options \NexFunction{opt}, based on the mapping provided in \NexFunction{fld}. Uses auxiliary Awk scripts for splitting parameters and retrieving structured outputs. Outputs formatted variables with prefix indicators like \texttt{GDF\_SL\_}.}
			\NexItemDark{\NexFunction{\_\_get\_dialog\_size()}: Retrieves the current terminal dimensions, including the number of rows and columns, for adaptive dialog layouts.}
			\NexItemDark{\NexFunction{get\_dialog\_explorer()}: Opens an interactive dialog window for file or directory exploration. Supports multiple dialog types (\texttt{fselect}, \texttt{dselect}, \texttt{textbox}, \texttt{editbox}, \texttt{tailboxbg}, \texttt{tailbox}) based on the provided input and conditions.}
			\NexItemDark{\NexFunction{get\_dialog\_yn()}: Displays a yes/no or message box dialog. Dynamically determines the type (\texttt{yesno} or \texttt{msgbox}) based on the provided options and user interaction requirements.}
			\NexItemDark{\NexFunction{get\_dialog\_cal()}: Launches a calendar dialog for date selection. Automatically formats the output to display the current date or allows custom configurations.}
			\NexItemDark{\NexFunction{get\_dialog\_time()}: Launches a dialog window to manage time-based interactions. Supports \NexFunction{pause} for countdowns and \NexFunction{timebox} for specific time selection, with configurable defaults and user-defined inputs.}
			\NexItemDark{\NexFunction{get\_dialog\_form()}: Generates a form-based dialog using dynamically formatted options and parameters. Processes user input via \NexFunction{\_\_get\_dialog\_factory} and extracts output using \NexFunction{\_\_get\_dialog\_output}. Returns the dialog’s exit status.}
			\NexItemDark{\NexFunction{get\_dialog\_menu()}: Constructs a menu-based dialog with options for various styles (e.g., \texttt{radiolist}, \texttt{checklist}). Utilizes \NexFunction{\_\_get\_dialog\_factory} for string formatting and \NexFunction{\_\_get\_dialog\_selected} for processing selections. Returns the dialog’s exit status.}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
