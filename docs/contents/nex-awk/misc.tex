\section{Miscellaneous}
\label{Miscellaneous}
\begin{NexMainBox}[light, hdrA, sdwA, crnA, grwB, secA]
	\begin{NexMainBox}[dark, crnA]
		The following functions provide generic utilities for handling values and conditional operations, offering flexible solutions for various logical scenarios.
	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{__return_value(D1, D2)}: Returns \NexFunction{D1} if it exists; otherwise, returns \NexFunction{D2}.}
			\NexItemDark{\NexFunction{__return_if_value(D1, D2, B)}: Combines \NexFunction{D1} and \NexFunction{D2} based on \NexFunction{B} and returns the result if \NexFunction{D1} exists.}
			\NexItemDark{\NexFunction{__return_else_value(D1, D2, B)}: Returns \NexFunction{D2} if \NexFunction{D1} satisfies the \NexFunction{TRUE__} condition for \NexFunction{B}.}
			\NexItemDark{\NexFunction{__load_value(V, K, DA, DB)}: Assigns \NexFunction{DA} to \NexFunction{V[K]} if \NexFunction{DA} is not "NULL"; otherwise, assigns \NexFunction{DB}.}
			\NexItemDark{\NexFunction{__load_delim(V, S, O)}: Sets delimiters in \NexFunction{V} by assigning \NexFunction{S} and \NexFunction{O} to keys "s" and "o" with defaults "," and newline.}
			\NexItemDark{\NexFunction{__load_tag(V, L, R)}: Sets tag delimiters in \NexFunction{V} by assigning \NexFunction{L} and \NexFunction{R} to keys "l" and "r" with defaults "<" and ">".}
		\end{NexListDark}
	\end{NexMainBox}
\end{NexMainBox}
