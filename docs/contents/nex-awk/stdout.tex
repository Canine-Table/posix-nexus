	\end{NexMainBox}
	\begin{NexMainBox}[dark, crnA]
		\begin{NexListDark}
			\NexItemDark{\NexFunction{load_symbols(V)}: Populates the array \NexFunction{V} with symbolic representations for text formatting or display purposes, using predefined mappings.}
			\NexItemDark{\NexFunction{text_style_map(D)}: Maps text style descriptors (\NexFunction{D}) like “bold” or “underlined” to their corresponding numerical codes.}
			\NexItemDark{\NexFunction{color_map(D)}: Maps color names or descriptors (\NexFunction{D}) like “red” or “warning” to their corresponding numerical codes for display purposes.}
		\end{NexListDark}
	\end{NexMainBox}
