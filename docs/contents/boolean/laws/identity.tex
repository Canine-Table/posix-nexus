\begin{ColorThemedBox}[1][2][title=Identity Law]\relax
	\begin{Items}
		\ItemArrow{\bftext{Identity Law for OR (Addition):}}

		\bftext{Expression:}\smallskip\\ $ A + 0 = A $

		\bftext{Explanation:} This means that if you OR (add) any Boolean variable $A$
		with 0 (false), the result is the original variable $A$
 		In other words, OR-ing with 0 does not change the value of $A$
		
		\ItemArrow{\bftext{Identity Law for AND (Multiplication)}:}
			
		\bftext{Expression:}\smallskip\\ $ A \cdot 1 = A $

		\bftext{Explanation:} Explanation: This means that if you AND (multiply) any Boolean variable $A$with 1 (true),
			the result is the original variable $A$. In other words, AND-ing with 1 does not change the value of $A$\smallskip
	\end{Items}
\end{ColorThemedBox}
