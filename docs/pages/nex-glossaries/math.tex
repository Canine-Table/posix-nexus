
\makeglossaries

\newglossaryentry{coefficient}{
	name={Coefficient},
  description={The numerical factor that multiplies a variable in a term}
}

\newglossaryentry{likeTerms}{
  name=Like Terms,
  description={Terms that have the same variable(s) raised to the same power(s)}
}

\newglossaryentry{numerator}{
  name={Numerator},
  description={the top number in a fraction, showing how many parts are taken}
}

\newglossaryentry{denominator}{
  name={Denominator},
  description={The bottom number in a fraction, showing how many equal parts make the whole}
}

\newglossaryentry{binomial}{
    name={Binomial},
    description={A polynomial with two terms. See equation~\ref{eq:binomial}.}
}

\newglossaryentry{divisionRule}{
    name={Division Rule},
    description={Subtract exponents when dividing like bases. Example: see equation~\ref{eq:division-basic}.}
}
