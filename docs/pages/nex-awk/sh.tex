\nxSections{Schema Compiler}{2}

\begin{empheq}[box=\nxWarningMathBox]{align*}
S_{\text{base}} &= 12
		&& \text{(fixed slots for keywords, flags, arrays)} \\
S_{\text{group}} &= 3
		&& \text{(leader type, leader symbol, end index)} \\
G &= \text{number of groups declared in the schema} \\
S_{\text{groups}} &= G \cdot S_{\text{group}}
		&& \text{(each group contributes exactly 3 slots)} \\
S_{\text{total}} &= S_{\text{base}} + S_{\text{groups}}
		&& \text{(final stride for all category walks)}
\end{empheq}
\bigskip

\begin{NxCodeBox}{awk}{phantomlabel={func:nx_shell_opts}, title={\textbf{Schema Compiler Arguments}
	\begin{NxDarkList}
		\nxEachArw{CodePrimary}{Tertiary}{1}{%
			{D1}/{Option schema string to tokenize (raw input)},%
			{V}/{Array to populate with parsed flags, keywords, arrays, groups},%
			{D2}/{Separator of separators (defaults to comma)},%
			{N}/{Runtime toggles string \nxSQ{dbg,ovr}},%
			{D3}/{List of separator characters (key, array, group, long/short mode)}%
		}
	\end{NxDarkList}
}}
function nx_shell_opts(D1, V, D2, N, D3)
\end{NxCodeBox}

		\hline

		\nxSQ{2} & ovr & Override mode for group type resolution (see below) \\

\end{NxLightBox}

\begin{NxLightListBox}[title={\texttt{ovr} semantics (group type override)}]
	\nxEachArw{ArrowDark}{Secondary}{1}{%
		{Default}/{When \texttt{ovr = 0}, existing group type is preserved when extending a group},%
		{Override}/{When \texttt{ovr = 1}, the group type symbol is replaced with the new one},%
		{Code path}/{On group extension, \texttt{ovr} controls whether \texttt{V[tmpb + strde]} is updated or left intact}%
	}
\end{NxLightListBox}

\begin{NxLightListBox}[title={debug levels}]
	\nxEachArw{BugSuccess}{Success}{1}{%

		{>0}/{errors only},%
		{>1}/{warnings},%
		{>2}/{verbose audit},%
		{>3}/{mapping summary},%
		{>4}/{full dump}%
	}%
\end{NxLightListBox}

\begin{NxLightBox}[title={Variable Positions in D3}, phantomlabel={func:nx_shell_opts:D3}]
	\begin{tabularx}{\textwidth}{|Z|c|Z|l|}
		\hline
		\textbf{Index} & \textbf{Default} & \textbf{Variable} & \textbf{Meaning} \\
		\hline
		- & \nxSQ{,} & ds & Default delimiter string \\
		1 & \nxSQ{:} & ks & Key separator \\
		2 & \nxSQ{@} & fas	& Appendable flag array separator \\
		3 & \nxSQ{\#} & kas	& Appendable keyword array separator \\
		4 & \nxSQ{<} & go	 & Begin group marker \\
		5 & \nxSQ{>} & gc	 & End group marker \\
		6 & \nxSQ{~~} & lo & Begin or continue long option mode \\
		7 & \nxSQ{;} & lc	 & End long option mode \\
		\hline
	\end{tabularx}
\end{NxLightBox}

\begin{NxLightListBox}[title={Short vs Long Option Group Leaders}]
		\nxEachArw{ArrowDark}{Secondary}{1}{%
			{Short form}/{Input: \nxSQ{\nxSpaces{alpha<beta gamma>}} \nxArw{2} Flags = a, l, p, h; Group leader = a; Members = b, e, t, a, gamma},%
				{Long form}/{Input: \nxSQ{\nxSpaces{ alpha<beta gamma>}} \nxArw{2} Group leader = alpha; Members = beta, gamma},%
				{Key distinction}/{Leading space (\nxSQ{lo}) preserves token as long option; without space, token explodes into flags and first flag becomes leader}%
		}
\end{NxLightListBox}

		Flag arrays			& 7	& $+\,S_{\text{total}}$ \\
		Keyword arrays	 & 10 & $+\,S_{\text{total}}$ \\

\end{NxLightBox}

\begin{NxLightListBox}[title={Schema Compiler --- Internal Semantics}]
	\nxEachArw{ArrowDark}{Secondary}{1}{%
		{Purpose}/{Compile schema string into a stride-aligned registry},%
		{Categories}/{Registers keywords, flags, flag arrays, keyword arrays, and groups},%
		{Stride}/{Computes total stride $S_{\text{total}} = 12 + S_{\text{groups}}$},%
		{Groups}/{Leader + members stored in triplets beginning at index 13},%
		{Long Option Mode}/{Preserves multi-character tokens when \texttt{lo} is active},%
		{Validation}/{Ensures separators are single, non-alphanumeric characters},%
		{Output}/{Populates registry \texttt{V} with canonical positions and namespace anchors}%
	}
\end{NxLightListBox}

\begin{NxLightListBox}[title={Category Anchors (Base Indices)}]
	\nxEachArw{ArrowDark}{Secondary}{1}{%
		{Keywords}/{Base index 1, stride $+S$},%
		{Flags}/{Base index 4, stride $+S$},%
		{Flag Arrays}/{Base index 7, stride $+S$},%
		{Keyword Arrays}/{Base index 10, stride $+S$},%
		{Groups}/{Base index 13, triplets spaced by 3, values spaced by $S$}%
	}
\end{NxLightListBox}

\nxSections{Examples}{3}

\begin{NxCodeBox}{awk}{title={\textbf{Register simple flags}%
	\begin{NxDarkList}
		\nxEachArw{CodePrimary}{Tertiary}{1}{%
			{State}/{\texttt{D1=\nxSQ{abc}}, \texttt{V=\{\}}, \texttt{N=\nxSQ{3,0}}},%
			{Operation}/{Compile schema into flag entries},%
			{Expected Result}/{\texttt{a}, \texttt{b}, \texttt{c} registered as flags at base index 4, spaced by $S_{\text{total}}$}
		}
	\end{NxDarkList}
}}
nx_shell_opts("abc", V, ",", "3,0", "")
\end{NxCodeBox}

\begin{NxCodeBox}{awk}{title={\textbf{Register keyword arguments (short form)}%
	\begin{NxDarkList}
		\nxEachArw{CodePrimary}{Tertiary}{1}{%
			{State}/{\texttt{D1=\nxSQ{k:x:y}}, \texttt{V=\{\}}, \texttt{N=\nxSQ{3,0}}},%
			{Operation}/{Use \texttt{ks = :} to mark keyword positions},%
			{Expected Result}/{\texttt{k}, \texttt{x}, \texttt{y} registered as keyword entries at base index 1, spaced by $S_{\text{total}}$}
		}
	\end{NxDarkList}
}}
nx_shell_opts("k:x:y", V, ",", "3,0", "")
\end{NxCodeBox}

\begin{NxCodeBox}{awk}{title={\textbf{Register long-form keyword}%
	\begin{NxDarkList}
		\nxEachArw{CodePrimary}{Tertiary}{1}{%
			{State}/{\texttt{D1=\nxSQ{ alpha:beta}}, \texttt{V=\{\}}, \texttt{N=\nxSQ{3,0}}},%
			{Operation}/{Leading space (\texttt{lo}) preserves \texttt{alpha} as a single token},%
			{Expected Result}/{\texttt{alpha} registered as a keyword entrie at base index 1 \texttt{beta} as a flag at base index 4}
		}
	\end{NxDarkList}
}}
nx_shell_opts(" alpha:beta", V, ",", "3,0", "")
\end{NxCodeBox}

\begin{NxCodeBox}{awk}{title={\textbf{Register flag arrays}%
	\begin{NxDarkList}
		\nxEachArw{CodePrimary}{Tertiary}{1}{%
			{State}/{\texttt{D1=\nxSQ{a@b@c}}, \texttt{V=\{\}}, \texttt{N=\nxSQ{3,0}}},%
			{Operation}/{Use \texttt{fas = @} to mark appendable flag-array entries},%
			{Expected Result}/{\texttt{a}, \texttt{b}, \texttt{c} registered at base index 7, spaced by $S_{\text{total}}$}
		}
	\end{NxDarkList}
}}
nx_shell_opts("a@b@c", V, ",", "3,0", "")
\end{NxCodeBox}

\begin{NxCodeBox}{awk}{title={\textbf{Register short-form option group}%
	\begin{NxDarkList}
		\nxEachArw{CodePrimary}{Tertiary}{1}{%
			{State}/{\texttt{D1=\nxSQ{a<b c>}}, \texttt{V=\{\}}, \texttt{N=\nxSQ{3,0}}},%
			{Operation}/{Group leader \texttt{a} with members \texttt{b}, \texttt{c}},%
			{Expected Result}/{Group header triplet starts at index 13; members stored at group base + $S_{\text{total}}$ multiples}
		}
	\end{NxDarkList}
}}
nx_shell_opts("a<b c>", V, ",", "3,0", "")
\end{NxCodeBox}

\begin{NxCodeBox}{awk}{title={\textbf{Register long-form option group}%
	\begin{NxDarkList}
		\nxEachArw{CodePrimary}{Tertiary}{1}{%
			{State}/{\texttt{D1=\nxSQ{ alpha<beta gamma>}}, \texttt{V=\{\}}, \texttt{N=\nxSQ{3,0}}},%
			{Operation}/{Leading space preserves \texttt{alpha} as group leader; \texttt{beta}, \texttt{gamma} as members},%
			{Expected Result}/{Group leader \texttt{alpha} stored in group header; members spaced by $S_{\text{total}}$}
		}
	\end{NxDarkList}
}}
nx_shell_opts(" alpha<beta gamma>", V, ",", "3,0", "")
\end{NxCodeBox}
