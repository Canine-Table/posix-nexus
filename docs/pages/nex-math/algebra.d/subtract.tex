\begin{NxLightListBox}[title={Minuend}]
	\nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
		{Definition}/{The number from which another is subtracted},%
		{Position}/{Always appears first in subtraction},%
		{Example}/{In \(9 - 4\), 9 is the minuend},%
		{Connection}/{Glyph of the “starting quantity”},%
		{Extension}/{Represents the whole or larger value in comparison}%
	}
\end{NxLightListBox}

\begin{NxLightListBox}[title={Subtrahend}]
	\nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
		{Definition}/{The number that is subtracted from the minuend},%
		{Position}/{Always appears second in subtraction},%
		{Example}/{In \(9 - 4\), 4 is the subtrahend},%
		{Connection}/{Glyph of the “taken away” quantity},%
		{Extension}/{Represents the part removed or compared against}%
	}
\end{NxLightListBox}
