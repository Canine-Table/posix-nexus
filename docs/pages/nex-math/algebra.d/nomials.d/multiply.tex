\begin{NxLightListBox}[title={Multiplication Symbols}]
    \nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
        {Dot}/{\(\cdot\) is clean, algebraic, avoids confusion with \(x\)},%
        {Times}/{\(\times\) is bold, arithmetic, or cross product},%
        {Context}/{Use \(\cdot\) in algebra, \(\times\) in arithmetic or vectors},%
        {Technique}/{Choose based on clarity and audience},%
        {Outcome}/{Both mean multiplication, but notation signals intent}%
    }
\end{NxLightListBox}

\begin{empheq}[box=\nxWarningMathBox]{align*}
	x^2 \cdot x^4 = x^{2 + 4} \quad & \nxArw{5} \; x^6
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	x^7 \cdot x^5 = x^{5 + 7} \quad & \nxArw{5} \; x^{12}
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	x^8 \cdot x^9 = x^{8 + 9} \quad & \nxArw{5} \; x^{17}
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	(3x^3)(5x^6) = (3 \cdot 5)x^{3 + 6} \quad & \nxArw{5} 15x^9
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	(4x^2)(7x^3) = (4 \cdot 7)x^{2 + 3} \quad & \nxArw{5} 28x^5
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	(4xy^2)(8x^2y^3) = (4 \cdot 8)x^{1 + 2}(1 \cdot 1)y^{2 + 3} \quad & \nxArw{5} \; 32x^3y^5 \\
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	(5x^2y^3)(6x^3y^4) = (5 \cdot 6)x^{2 + 3}(1 \cdot 1)y^{3 + 4} \quad & \nxArw{5} \; 30x^5y^7 \\
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	(7x^3y^4)(8x^5y^7) = (7 \cdot 8)x^{3 + 5}(1 \cdot 1)y^{4 + 7} \quad & \nxArw{5} \; 56x^8y^{11} \\
\end{empheq}
