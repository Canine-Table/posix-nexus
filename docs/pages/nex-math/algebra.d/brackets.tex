        {Programming}/{Common in code to enclose blocks of statements},%
        {\LaTeX}/{Used to group arguments for commands},%
        {Example}/{\(\{x \mid x > 0\}\) means the set of all positive \(x\)},%
        {Outcome}/{Curly braces signal structured grouping or set notation}%
    }
\end{NxLightListBox}

\begin{NxLightListBox}[title={Square Brackets []}]
    \nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
        {Definition}/{Used for intervals, optional elements, or matrices},%
        {Interval}/{\([a,b]\) means all values between \(a\) and \(b\), inclusive},%
        {Matrix}/{Brackets often enclose arrays of numbers},%
        {Example}/{\([2,5]\) includes 2 and 5},%
        {Outcome}/{Square brackets emphasize inclusion or structured arrays}%
    }
\end{NxLightListBox}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	8[−6 + 8(−2 + 4)] &= 8[-6 + 8 \cdot 2] \\
										&= 8[-6 + 16] \\
										&= 8 \cdot 10 \\
										&= 80
\end{empheq}

\begin{NxLightListBox}[title={Parentheses ()}]
    \nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
        {Definition}/{Used for grouping, order of operations, or function arguments},%
        {Math}/{\((a+b)c\) ensures addition happens before multiplication},%
        {Functions}/{\(f(x)\) shows input to a function},%
        {Interval}/{\((a,b)\) means values strictly between \(a\) and \(b\)},%
        {Outcome}/{Parentheses control grouping and precedence in math and logic}%
    }
\end{NxLightListBox}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	-(-(1-2) + (5)) &= -(-(-1) + 5) \\
									&= -(1 + 5) \\
									&= -(6) \\
									&= -6
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	3 \cdot (6 - 3 + 1) - 4^2 &= 3 \cdot 4 - 16 \\
														&= 12 - 16 \\
														&= -4
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	(10,4) \nxArw{18} \gcd(10,4) = 2 & \\
	\frac{2 \cdot (5 - 1) + 2}{4 \cdot (2 - 1)} & \;\nxArw{22}\; (2 \cdot (5 - 1) + 2) \div (4 \cdot (2 - 1)) \\
																							&= \frac{2 \cdot 4 + 2}{4\cdot 1} \\
																							&= \frac{8 + 2}{4} \\
																							&= \frac{10}{4} \\
																							&= \frac{\frac{10}{2}}{\frac{4}{2}} \\
																							&= \frac{5}{2}
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	5 + (1 \cdot 4 - (2 - 5) - 2) &= 4 - (-3) - 2 \\
																&= 7 - 2 \\
																&= 5
\end{empheq}
\bigskip

\begin{NxLightListBox}[title={Double Vertical Bars ||}]
    \nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
        {Definition}/{Used to denote absolute value or norm},%
        {Absolute Value}/{\(|x|\) is distance of \(x\) from 0},%
        {Norm}/{\(\|v\|\) is length of vector \(v\)},%
        {Example}/{\(|-5| = 5\), \(\| (3,4) \| = 5\)},%
        {Outcome}/{Double bars measure magnitude or distance}%
    }
\end{NxLightListBox}
\begin{empheq}[box=\nxWarningMathBox]{align}
	4 + |3-9| &= 4 + |-6| \\
						&= 4 + 6 \\
						&= 10
\end{empheq}
\bigskip
