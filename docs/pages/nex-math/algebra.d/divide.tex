\begin{empheq}[box=\nxWarningMathBox]{align}
	\frac{x^8}{x^3} = x^{8 - 3} & \nxArw{5} x^5 \label{eq:division-basic} \\
		\frac{x^5}{x^2} = x^{5 - 2} & \nxArw{5} x^3 \\
		\frac{x^5}{x^8} = x^{5 - 8} & \nxArw{5} x^{-3} \label{eq:division-negative} \\
	\frac{x^4}{x^7} = x^{4 - 7} & \nxArw{5} x^{-3} \\
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	\frac{24x^9y^5}{8x^3y^12} &= \frac{24}{8}x^{9 - 3}\frac{1}{1}y^{5 - 12} \\
														&= 3x^6y^{-7} \\
														&=\frac{3x^6}{y^7}
\end{empheq}
\begin{empheq}[box=\nxWarningMathBox]{align*}
	\frac{12x^5y^{-3}z^4}{36x^8y^{-4}x^{-8}} &= \frac{\frac{12}{3}x^{5 - 8}}{\frac{36}{3}}\frac{1}{1}y^{-3 - -4}\frac{1}{1}z^{4 - -8} \\
																					 &= \frac{x^{-3}y^1z^{12}}{3} \\
																					 &= \frac{yz^{12}}{3x^3}
\end{empheq}
