\begin{NxLightListBox}[title={FOIL Method}]
    \nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
        {First}/{Multiply first terms: \(a \cdot c\)},%
        {Outer}/{Multiply outer terms: \(a \cdot d\)},%
        {Inner}/{Multiply inner terms: \(b \cdot c\)},%
        {Last}/{Multiply last terms: \(b \cdot d\)},%
        {Outcome}/{Sum all four products to get the expanded expression}%
    }
\end{NxLightListBox}

\begin{empheq}[box=\nxWarningMathBox]{align*}
	(2x+5)(4x^2-3x+6)
		= (2 \cdot 4)x^{1 + 2}
		+ (2 \cdot -3)^{1 + 1}
		+ (2 \cdot 6)x \\
		\nxArw{20}
		+ (5 \cdot 4)x^2
		+ (5 \cdot -3)x
		+ (5 \cdot 6) \\
		= 8x^3
		+ (-6 + 20)^2
		+ (12 + -15)x
		+ 30 \\
		= 8x^3
		+ 14x^2
		+ -3x
		+ 30
\end{empheq}
