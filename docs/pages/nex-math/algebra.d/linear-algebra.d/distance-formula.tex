\begin{NxLightListBox}[title={Distance Formula}]
    \nxEachLabel{ArrowDark}{Secondary}{{4}{5}}{%
        {Definition}/{Gives the length of the line segment between two points in the plane},%
        {Points}/{Two points \((x_1,y_1)\) and \((x_2,y_2)\)},%
        {Formula}/{\(\sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}\)},%
        {Origin}/{Derived from the Pythagorean Theorem},%
        {Domain}/{Applies to all real coordinates in 2D space},%
        {Extension}/{Generalizes to 3D: \(\sqrt{(x_2-x_1)^2+(y_2-y_1)^2+(z_2-z_1)^2}\)}%
    }
\end{NxLightListBox}

\begin{empheq}[box=\nxSuccessMathBox]{align}
	\text{2D points} \qquad d = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2} \\
	\text{2D points} \qquad d = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2 + (z_2 - z_1)^2}
\end{empheq}
