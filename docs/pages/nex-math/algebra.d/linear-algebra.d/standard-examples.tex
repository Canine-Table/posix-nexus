\nxLnArcBox[]{\bf\nxSFont{12pt}$3x - 2y = 6$}
\begin{empheq}[box=\nxLightMathBox]{align*}
	3x - 2y = 6 \\
	\frac{3x}{3} - 2y \times 0 = \frac{6}{3} \\
	x = 2 \\
	3x \times 0 - \frac{2y}{-2} = \frac{6}{-2} \\
	y = -3 \\
	\text{(2, 0), (0, -3)}
	slope = \frac{0 - -3}{2 - 0} = \frac{3}{2}
\end{empheq}
\bigskip

\nxLnArcBox[]{\bf\nxSFont{12pt}$4x + 3y = 12$}
\begin{empheq}[box=\nxLightMathBox]{align*}
	4x - 3y = 12 \\
	\frac{4x}{4} + 3y \times 0 = \frac{12}{4} \\
	x = 3 \\
	4x \times 0 + \frac{3y}{3} = \frac{12}{3} \\
	y = 4 \\
	\text{(3, 0), (0, 4)} \\
	slope = \frac{y_2 - y_1}{x_2 - x_1}  = \frac{0 - 4}{3 - 0} = \frac{-4}{3} \\
\end{empheq}
\bigskip
