\begin{NxCodeBox}{sql}{title={\textbf{Creating a New Application User}
        \begin{NxDarkList}
                \nxEachLabel{DBSecondary}{Tertiary}{{1}{43}}{%
                        {app_user}/{Represents a database identity with login credentials and access privileges.},%
                        {user creation}/{Defines authentication method, resource limits, and optional role membership.}
                }
        \end{NxDarkList}
}}

-- PostgreSQL
CREATE USER app_user
WITH PASSWORD 'securepass123'
CREATEDB
LOGIN
CONNECTION LIMIT 10;

-- MariaDB / MySQL
CREATE USER 'app_user'@'localhost'
IDENTIFIED BY 'securepass123'
PASSWORD EXPIRE INTERVAL 90 DAY
COMMENT 'Application-level user with limited access';

-- SQLite3
-- No CREATE USER command; access control is handled externally
-- Example: authentication managed by application or file system
-- Users are simulated via application logic
\end{NxCodeBox}
