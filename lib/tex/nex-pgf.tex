\def\nx@Crowsfoot@Setup@{%
    \pgfarrowssettipend{\nx@tipend\pgfarrowlength}
    \pgfarrowssetlineend{-\nx@lineend\pgfarrowlength}
    \pgfarrowssetvisualbackend{\nx@visualbackend\pgfarrowlength}
    \pgfarrowssetbackend{-\nx@backend\pgfarrowlength}
    \pgfarrowshullpoint{\nx@tipend\pgfarrowlength}{\nx@tipoffset}
    \pgfarrowshullpoint{-\nx@backend\pgfarrowlength}{\nx@tipend\pgfarrowlength}
    \pgfarrowshullpoint{-\nx@backend\pgfarrowlength}{-\nx@tipend\pgfarrowlength}
    \pgfarrowssavethe\pgfarrowlength
}
\def\nx@Crowsfoot@Circle@Fill{%
	\pgfpathcircle{\pgfpoint{\nx@circleoffset\pgfarrowlength}{\nx@circlestart}}{\nx@circleradius\pgfarrowlength}%
	\pgfpathclose%
	\pgfusepathqfill%
}

}
