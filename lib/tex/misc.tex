\def\EncodingMap{%
	\AddCount{encodingmap}{0}\par%
	\loop\unless\ifnum\the\encodingmap>127%
		[\the\encodingmap =\char\the\encodingmap]%
		\AddCount{encodingmap}{1}%
	\repeat\par%
}

\def\SnowFlakeHeader{%
	\global\renewcommand\headrule{%
		\vspace{-0.75in}%
		\def\SnowFlake{%
			\tikz[decoration=Koch snowflake]{%
				\draw decorate{%
					decorate{%
						decorate{(0,-2) -- (3,-2)}%
					}%
				};%
			}%
		}%
		\ifodd\the\count0%
			\hrulefill{\SnowFlake}%
		\else%
			\SnowFlake\hrulefill{}%
		\fi%
	}
}
