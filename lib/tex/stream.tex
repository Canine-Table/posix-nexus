\def\ReadFile#1{%
	\Group{%
		\def\TmpA ##1{%
			\def\InLineNoSpace{##1}%
		}%
		\newread\stream%
		\openin\stream={#1}%
		\loop\unless\ifeof\stream%
			\read\stream to\InLine%
			\InLine\par
		\repeat%
		\closein\stream%
	}%
}

\long\def\WriteFile#1#2{%
	\Group{%
		\newwrite\stream%
		\openout\stream={#1}%
		\immediate\write\stream{#2}%
		\closeout\stream%
	}%
}
