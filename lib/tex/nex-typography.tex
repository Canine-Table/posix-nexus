\def\nxEm#1{%
	\edef\nx@tmpb{\nxDigit{#1}}%
	\ifnum\nx@tmpb>-1
		\expandafter\csname
			\ifcase#1
				uline% 0
			\or
				uwave% 1
			\or
				sout% 2
			\or
				xout% 3
			\or
				dashuline% 4
			\else
				dotuline% 5
			\fi
		\expandafter\endcsname
	\fi
}

\def\nxSz#1{%
	\edef\nx@tmpb{\nxDigit{#1}}%
	\ifnum\nx@tmpb>-1
		\expandafter\csname
			\ifcase#1
				tiny% 0
			\or%
				scriptsize% 1
			\or%
				footnotesize% 2
			\or%
				small% 3
			\or%
				normalsize% 4
			\or%
				large% 5
			\or%
				Large% 6
			\or%
				LARGE% 7
			\or%
				huge% 8
		\else%
				Huge% >= 9
			\fi%
		\endcsname
	\fi
}
