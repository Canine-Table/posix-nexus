\def\NewCount#1{\SelectRegister{#1}1\PickFirstOfOne}
\def\ExprCount#1{\the\numexpr#1\relax}
\def\AddCount#1{\UseCount{advance}{#1}}
\def\MultiCount#1{\UseCount{multiply}{#1}}
\def\DivCount#1{\UseCount{divide}{#1}}
\def\UseCount#1#2{%
	\CheckDef{#2}{%
		\expandafter\csname#1\endcsname\csname#2\endcsname by\expandafter\PickFirstOfOne%
	}{%
		\NewCount{#2}%
	}%
}
